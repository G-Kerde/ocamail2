Module Ocamail.StrUtils


module StrUtils: sig .. end
Some String utilities
val lines_of : string -> string list
Returns a list containing each line in the given string
val header_parts : string -> string * string
Return a tuple where fst is the key and snd is the value in a string expression like "key: value"
val trim : string -> string
trim a string by removing "^[ \t\r\n]+" and "[ \t\r\n]+$" from it. Example trim "\thello\t \tworld ! \t" == "hello\t \tworld !"
val after_colon : string -> string
Returns what is after a colon, in the given line
val safe_char : int -> string
Get the String representation of a char code, even if it exceeds unsigned char limits. Works as C unsigned char casting.
